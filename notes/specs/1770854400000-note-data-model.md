## Note 型

```ts
interface Note {
  id: string           // タイムスタンプ-スラッグ
  slug: string         // category/id
  title: string
  content: string
  status: NoteStatus
  category: NoteCategory
  ai_outcome: AiOutcome
  ai_reviewed: boolean
  createdAt: string    // ISO 8601
  updatedAt: string    // ISO 8601
  sha?: string         // GitHub blob SHA（競合検知用）
}
```
